### Code implementation
```Java
// Class Person
public class Person {
    private String name;
    private Car car; // Association: A person can have a car

    public Person(String name) {
        this.name = name;
    }

    public void setCar(Car car) {
        this.car = car;
    }

    public Car getCar() {
        return car;
    }

    public String getName() {
        return name;
    }
}

// Class Car
public class Car {
    private String model;

    public Car(String model) {
        this.model = model;
    }

    public String getModel() {
        return model;
    }
}
```
